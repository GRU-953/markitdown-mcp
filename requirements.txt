# MarkItDown Attachments MCP server — Python dependencies.
# Targets Python 3.10–3.13 (markitdown's supported range). On 3.14 some wheels
# (e.g. youtube-transcript-api, onnxruntime) may be unavailable; install.sh
# prefers a 3.12 interpreter for this reason.
mcp~=1.8.0
# File-attachment extras only (PDF, Word, Excel, PowerPoint, Outlook .msg, audio).
# Deliberately excludes the YouTube + Azure cloud extras from markitdown[all].
# (pdfminer.six + pypdfium2 come via the pdf extra; pypdfium2 is reused for OCR rasterization.)
markitdown[pdf,docx,xlsx,xls,pptx,outlook,audio-transcription]>=0.1.1,<0.2.0
# OCR for images & scanned PDFs. Requires the Tesseract *system* binary too:
#   macOS:  brew install tesseract   (and `brew install tesseract-lang` for non-English)
#   Debian: apt-get install tesseract-ocr
# Without Tesseract the server still runs — it just reports OCR as unavailable.
pytesseract>=0.3.10
# Local, offline speech-to-text for audio (replaces markitdown's cloud Google transcription).
faster-whisper>=1.0.0
# Plain-text extraction for legacy RTF (markitdown has no RTF converter).
striprtf>=0.0.26

# Optional — local image *descriptions* (vision) for images OCR can't read.
# Uses the Ollama app + an open-source vision model. No extra Python package is
# needed ('requests', already pulled in by markitdown, calls Ollama over localhost):
#   macOS:  brew install ollama && ollama pull moondream
# The server starts Ollama on demand and stops it after idle (OLLAMA_IDLE_TIMEOUT,
# default 300s) — no need to keep it running in the background.
